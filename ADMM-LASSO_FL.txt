
### Load the "glmnet" package
library(glmnet)

# Number of replications
num_replications <- 100

# Number of features
p <- 100

# Simulate data for two clients
n <- 200  # Number of samples per client
m <- 2   # Number of clients

# Sequence of lambda values for tuning
lambda_seq <- seq(0.01, 3, by = 0.1) 

# ADMM penalty parameter
rho <- 0.012

# Threshold to determine if the coefficient is active
threshold <- 0.001
# Absolute tolerance
eps_abs <- 1e-4
# Relative tolerance
eps_rel <- 1e-3

# Storage for ADMM-LASSO
# Store global beta from ADMM
all_global_beta <- matrix(0, nrow = num_replications, ncol = p)
TP_list <- FP_list <- FN_list <- TN_list <- numeric(num_replications)
sensitivity <- specificity <- numeric(num_replications)
MSE_ADMM_LASSO <- MAE_ADMM_LASSO <- MSE_PRED_ADMM_LASSO <- numeric(num_replications)

# Storage for LASSO-POOL
beta_pool_lasso <- matrix(0, nrow = num_replications, ncol = p)
TP_list_LP <- FP_list_LP <- FN_list_LP <- TN_list_LP <- numeric(num_replications)
sensitivity_LP <- specificity_LP <- numeric(num_replications)
MSE_LASSO_POOL <- MAE_LASSO_POOL <- MSE_PRED_LASSO_POOL <- numeric(num_replications)

# Storage for LASSO-1 
all_beta_client1 <- matrix(0, nrow = num_replications, ncol = p)
TP_list_L1 <- FP_list_L1 <- FN_list_L1 <- TN_list_L1 <- numeric(num_replications)
sensitivity_L1 <- specificity_L1 <- numeric(num_replications)
MSE_LASSO_1 <- MAE_LASSO_1 <- MSE_PRED_LASSO_1 <- numeric(num_replications)

######Storage for LASSO-2 ######
all_beta_client2 <- matrix(0, nrow = num_replications, ncol = p)
TP_list_L2 <- FP_list_L2 <- FN_list_L2 <- TN_list_L2 <- numeric(num_replications)
sensitivity_L2 <- specificity_L2 <- numeric(num_replications)
MSE_LASSO_2 <- MAE_LASSO_2 <- MSE_PRED_LASSO_2 <- numeric(num_replications)

# Soft-thresholding function (proximal operator for the L1 penalty) used in LASSO
soft_threshold <- function(x, lambda) {
  sign(x) * pmax(0, abs(x) - lambda)
}

# Function to perform local LASSO update in ADMM
admm_local_lasso <- function(X, y, z, u, rho) {
  
  # Solve for local clients
  p <- ncol(X)
  a=t(X) %*% X + rho * diag(p)
  b=(t(X) %*% y + rho * (z - u))
  beta <- solve(a) %*% b
  return(as.vector(beta))
}

####### Simulation Loop #########
#Simulates the process for a number of repetitions to assess performance stability.
for (i in 1:num_replications) {
  
  ## Set seed
  set.seed(100 + i)
  
  ### True beta
  true_beta <- c(5, -2, 3, 1.5, 7, rep(0, p - 5))
  
  ### Simulate the data from the standard normal distribution
  simulate_data <- function() {
    X <- matrix(rnorm(n * p), nrow = n)
    y <- X %*% true_beta + rnorm(n)
    return(list(X = X, y = y))
  }
  
  ### Simulate the Training Data
  clients <- lapply(1:m, function(i) simulate_data())
  
  ### Simulate the Validation data
  clients_val <- lapply(1:m, function(i) simulate_data())
  
  ### Simulate the test Data 
  clients_test <- lapply(1:m, function(i) simulate_data())
  
  #### Lambda Tuning for ADMM using the Validation data  ######### 
  val_errors <- numeric(length(lambda_seq))
  
  for (l in seq_along(lambda_seq)) {
    lambda_try <- lambda_seq[l]
    
    # Initialize global variable z and dual variables u
    z <- rep(0, p)
    u_list <- lapply(1:m, function(i) rep(0, p))
    
    for (iter in 1:1000) {
      beta_list <- list()
      
      #Each client performs a local update
      for (k in 1:m) {
        beta_list[[k]] <- admm_local_lasso(clients[[k]]$X, clients[[k]]$y, z, u_list[[k]], rho)
      }
      z_old <- z
      
      # Server aggregates (beta + u) and updates z
      avg_bu <- Reduce(`+`, Map(`+`, beta_list, u_list)) / m
      z_new  <- soft_threshold(avg_bu, lambda_try / (m * rho)) 
      
      # Clients update dual variable
      for (k in 1:m) {
        u_list[[k]] <- u_list[[k]] + (beta_list[[k]] - z_new)
      }
      
      # Residual calculations (stacked across workers)
      r_norm <- sqrt(sum(sapply(1:m, function(k) sum((beta_list[[k]] - z_new)^2))))
      s_norm <- rho * sqrt(m) * sqrt(sum((z_new - z_old)^2))
      
      beta_norm <- sqrt(sum(sapply(beta_list, function(b) sum(b^2))))
      z_norm <- sqrt(m) * sqrt(sum(z_new^2))
      u_norm <- sqrt(sum(sapply(u_list, function(u) sum(u^2))))
      
      # Tolerances (Boyd, scaled form)
      eps_primal <- sqrt(p * m) * eps_abs + eps_rel * max(beta_norm, z_norm)
      eps_dual <- sqrt(p*m) * eps_abs + eps_rel * (rho * u_norm)
      z <- z_new
      if (r_norm < eps_primal && s_norm < eps_dual) break
    }
    
    ### Calculate the MSE prediction using global model z and validation data ####
    # Store validation error
    val_error <- mean(sapply(1:m, function(k) {
      pred <- clients_val[[k]]$X %*% z
      mean((pred - clients_val[[k]]$y)^2)
    }))
    val_errors[l] <- val_error
  }
  
  ### Select the best lambda based on minimum error####
  best_lambda <- lambda_seq[which.min(val_errors)]
  lambda <- best_lambda
  
  ############################### Final ADMM with best lambda ######
  # initialize global variable z
  z <- rep(0, p)
  
  # Number of ADMM iterations
  num_iterations <- 1000
  
  ### Initialize the dual variable###
  u_list <- lapply(1:m, function(i) rep(0, p))
  
  # Federated learning loop from local update to global model  
  for (iter in 1:num_iterations) {
    ##### store beta from each client######
    beta_list <- list()
    for (k in 1:m) {
      # Perform local LASSO update for client k using current z and u_k
      beta_list[[k]] <- admm_local_lasso(clients[[k]]$X, clients[[k]]$y, z, u_list[[k]], rho)
    }
    
    z_old <- z
    
    # Server aggregates (beta + u) and updates z
    avg_bu <- Reduce(`+`, Map(`+`, beta_list, u_list)) / m
    z_new  <- soft_threshold(avg_bu, lambda / (m * rho)) 
    
    # Clients update dual variable
    for (k in 1:m) {
      u_list[[k]] <- u_list[[k]] + (beta_list[[k]] - z_new)
    }
    
    # Residual calculations (stacked across workers)
    r_norm <- sqrt(sum(sapply(1:m, function(k) sum((beta_list[[k]] - z_new)^2))))
    s_norm <- rho * sqrt(m) * sqrt(sum((z_new - z_old)^2))
    
    beta_norm <- sqrt(sum(sapply(beta_list, function(b) sum(b^2))))
    z_norm <- sqrt(m) * sqrt(sum(z_new^2))
    u_norm <- sqrt(sum(sapply(u_list, function(u) sum(u^2))))
    
    # Tolerances (Boyd, scaled form)
    eps_primal <- sqrt(p * m) * eps_abs + eps_rel * max(beta_norm, z_norm)
    eps_dual <- sqrt(p*m) * eps_abs + eps_rel * (rho * u_norm)
    z <- z_new
    if (r_norm < eps_primal && s_norm < eps_dual) break
  }
  
  ##### Store final global beta estimate after ADMM convergence
  global_beta <- z
  
  ### Global Beta #####
  all_global_beta[i, ] <- global_beta
  
  
  # Initialize counts for TP, FP, FN, TN
  TP <- 0
  FP <- 0
  FN <- 0
  TN <- 0
  
  # Loop through each beta coefficient to calculate TP, FP, FN, TN
  for (j in 1:length(true_beta)) {
    if (abs(true_beta[j]) > threshold) {  # True non-zero beta
      if (abs(global_beta[j]) > threshold) {
        TP <- TP + 1  # True non-zero estimated as non-zero
      } else {
        FN <- FN + 1  # True non-zero estimated as zero
      }
    } else {  # True zero beta
      if (abs(global_beta[j]) > threshold) {
        FP <- FP + 1  # True zero estimated as non-zero
      } else {
        TN <- TN + 1  # True zero estimated as zero
      }
    }
  }
  
  # Store the counts
  TP_list[i] <- TP
  FP_list[i] <- FP
  FN_list[i] <- FN
  TN_list[i] <- TN
  
  #Calculate Sensitivity (SE)
  sensitivity[i] <- TP / (TP + FN)
  
  #Calculate Specificity (SP)
  specificity[i] <- TN / (TN + FP)
  
  ####### MSE and MAE for ADMM-LASSO for coefficient estimation ##########
  MSE_ADMM_LASSO[i] <- mean((global_beta - true_beta)^2)
  MAE_ADMM_LASSO[i] <- mean(abs(global_beta - true_beta))
  
  ####### MSE for ADMM-LASSO predictions ##########
  pred1 <- clients_test[[1]]$X %*% global_beta
  pred2 <- clients_test[[2]]$X %*% global_beta
  y1 <- clients_test[[1]]$y
  y2 <- clients_test[[2]]$y
  MSE_PRED_ADMM_LASSO[i] <- (mean((pred1 - y1)^2) + mean((pred2 - y2)^2)) / m
  
  ################################ LASSO-POOL##################
  pooled_X <- do.call(rbind, lapply(clients, function(cl) cl$X))
  pooled_y <- do.call(c, lapply(clients, function(cl) cl$y))
  pooled_X_val <- do.call(rbind, lapply(clients_val, function(cl) cl$X))
  pooled_y_val <- do.call(c, lapply(clients_val, function(cl) cl$y))
  #### selecting the Best lambda
  mse_pool <- sapply(lambda_seq, function(lambda_pool) {
    fit <- glmnet(pooled_X, pooled_y, alpha = 1, lambda = lambda_pool)
    b <- as.vector(coef(fit))[-1]
    mean((pooled_X_val %*% b - pooled_y_val)^2)
  })
  best_lambda_pool <- lambda_seq[which.min(mse_pool)]
  
  ### Fitting lasso model for pool data with best lambda ###
  fit_pool <- glmnet(pooled_X, pooled_y, alpha = 1, lambda = best_lambda_pool)
  beta_pool <- as.vector(coef(fit_pool))[-1]
  ######## Lasso beta###
  beta_pool_lasso[i, ] <- beta_pool
  
  # Initialize counts for TP, FP, FN, TN
  TP_LP <- 0
  FP_LP <- 0
  FN_LP <- 0
  TN_LP <- 0
  # Loop through each beta coefficient to calculate TP, FP, FN, TN
  for (j in 1:length(true_beta)) {
    if (abs(true_beta[j]) > threshold) {  # True non-zero beta
      if (abs(beta_pool[j]) > threshold) {
        TP_LP <- TP_LP + 1  # True non-zero estimated as non-zero
      } else {
        FN_LP <- FN_LP + 1  # True non-zero estimated as zero
      }
    } else {  # True zero beta
      if (abs(beta_pool[j]) > threshold) {
        FP_LP <- FP_LP + 1  # True zero estimated as non-zero
      } else {
        TN_LP <- TN_LP + 1  # True zero estimated as zero
      }
    }
  }
  
  # Store the counts
  TP_list_LP[i] <- TP_LP
  FP_list_LP[i] <- FP_LP
  FN_list_LP[i] <- FN_LP
  TN_list_LP[i] <- TN_LP
  
  # Calculate Sensitivity (SE)
  SE_LP <- TP_LP / (TP_LP + FN_LP)
  sensitivity_LP[i] <- SE_LP
  
  # Calculate Specificity (SP)
  SP_LP <- TN_LP / (TN_LP + FP_LP)
  specificity_LP[i] <- SP_LP
  
  ####### MSE and MAE for LASSO-POOl coefficient estimation ##########
  MSE_LASSO_POOL[i] <- mean((beta_pool - true_beta)^2)
  MAE_LASSO_POOL[i] <- mean(abs(beta_pool - true_beta))
  
  ####### MSE for LASSO-POOL predictions ##########
  pred <- do.call(rbind, lapply(clients_test, function(cl) cl$X)) %*% beta_pool
  y <- do.call(c, lapply(clients_test, function(cl) cl$y))
  MSE_PRED_LASSO_POOL[i] <- mean((pred - y)^2)
  
  #################### LASSO-1########################
  #selecting best lambda
  mse_L1 <- sapply(lambda_seq, function(lambda_l1) {
    fit_cl_l1 <- glmnet(clients[[1]]$X, clients[[1]]$y, alpha = 1, lambda = lambda_l1)
    beta_cl_l1 <- as.vector(coef(fit_cl_l1))[-1]
    mean((clients_val[[1]]$X %*% beta_cl_l1 - clients_val[[1]]$y)^2)
  })
  best_lambda_L1 <- lambda_seq[which.min(mse_L1)]
  ### Fit lasso with the best lambda####
  fit_L1 <- glmnet(clients[[1]]$X, clients[[1]]$y, alpha = 1, lambda = best_lambda_L1)
  beta_client1 <- as.vector(coef(fit_L1))[-1]
  
  all_beta_client1[i, ] <- beta_client1
  
  # Initialize counts for TP, FP, FN, TN
  TP_L1 <- 0
  FP_L1 <- 0
  FN_L1 <- 0
  TN_L1 <- 0
  
  # Loop through each beta coefficient to calculate TP, FP, FN, TN
  for (j in 1:length(true_beta)) {
    if (abs(true_beta[j]) > threshold) {  # True non-zero beta
      if (abs(beta_client1[j]) > threshold) {
        TP_L1 <- TP_L1 + 1  # True non-zero estimated as non-zero
      } else {
        FN_L1 <- FN_L1 + 1  # True non-zero estimated as zero
      }
    } else {  # True zero beta
      if (abs(beta_client1[j]) > threshold) {
        FP_L1 <- FP_L1 + 1  # True zero estimated as non-zero
      } else {
        TN_L1 <- TN_L1 + 1  # True zero estimated as zero
      }
    }
  }
  
  # Store the counts
  TP_list_L1[i] <- TP_L1
  FP_list_L1[i] <- FP_L1
  FN_list_L1[i] <- FN_L1
  TN_list_L1[i] <- TN_L1
  
  # Calculate Sensitivity (SE)
  SE_L1 <- TP_L1 / (TP_L1 + FN_L1)
  sensitivity_L1[i] <- SE_L1
  
  # Calculate Specificity (SP)
  SP_L1 <- TN_L1 / (TN_L1 + FP_L1)
  specificity_L1[i] <- SP_L1
  
  ####### MSE and MAE for LASSO-1 coefficient estimation ##########
  MSE_LASSO_1[i] <- mean((beta_client1 - true_beta)^2)
  MAE_LASSO_1[i] <- mean(abs(beta_client1 - true_beta)) 
  
  ####### MSE for LASSO-1 predictions ##########
  pred_client1_1 <- clients_test[[1]]$X %*% beta_client1
  MSE_PRED_LASSO_1[i] <- mean((pred_client1_1 - clients_test[[1]]$y)^2)
  
  ################## LASSO-2###############
  mse_L2 <- sapply(lambda_seq, function(lambda_l2) {
    fit_cl_L2 <- glmnet(clients[[2]]$X, clients[[2]]$y, alpha = 1, lambda = lambda_l2)
    beta_cl_l2 <- as.vector(coef(fit_cl_L2))[-1]
    mean((clients_val[[2]]$X %*% beta_cl_l2 - clients_val[[2]]$y)^2)
  })
  best_lambda_L2 <- lambda_seq[which.min(mse_L2)]
  
  ### Fitting LASSO-2 with best lambda #####
  fit_L2 <- glmnet(clients[[2]]$X, clients[[2]]$y, alpha = 1, lambda = best_lambda_L2)
  
  beta_client2 <- as.vector(coef(fit_L2))[-1]
  all_beta_client2[i, ] <- beta_client2
  
  # Initialize counts for TP, FP, FN, TN
  TP_L2 <- 0
  FP_L2 <- 0
  FN_L2 <- 0
  TN_L2 <- 0
  # Loop through each beta coefficient to calculate TP, FP, FN, TN
  for (j in 1:length(true_beta)) {
    if (abs(true_beta[j]) > threshold) {  # True non-zero beta
      if (abs(beta_client2[j]) > threshold) {
        TP_L2 <- TP_L2 + 1  # True non-zero estimated as non-zero
      } else {
        FN_L2 <- FN_L2 + 1  # True non-zero estimated as zero
      }
    } else {  # True zero beta
      if (abs(beta_client2[j]) > threshold) {
        FP_L2 <- FP_L2 + 1  # True zero estimated as non-zero
      } else {
        TN_L2 <- TN_L2 + 1  # True zero estimated as zero
      }
    }
  }
  
  # Store the counts
  TP_list_L2[i] <- TP_L2
  FP_list_L2[i] <- FP_L2
  FN_list_L2[i] <- FN_L2
  TN_list_L2[i] <- TN_L2
  
  # Calculate Sensitivity (SE)
  SE_L2 <- TP_L2 / (TP_L2 + FN_L2)
  sensitivity_L2[i] <- SE_L2
  
  # Calculate Specificity (SP)
  SP_L2 <- TN_L2 / (TN_L2 + FP_L2)
  specificity_L2[i] <- SP_L2
  
  ####### MSE and MAE for LASSO-2 coefficient estimation ##########
  MSE_LASSO_2[i] <- mean((beta_client2 - true_beta)^2)
  MAE_LASSO_2[i] <- mean(abs(beta_client2 - true_beta))
  
  ####### MSE for LASSO-2 predictions ##########
  pred_client2_2 <- clients_test[[2]]$X %*% beta_client2
  MSE_PRED_LASSO_2[i] <- mean((pred_client2_2 - clients_test[[2]]$y)^2)
}
